def format_body(desc, subtasks = [])
  return desc if subtasks.nil? || subtasks.empty?

  [desc, format_checklist(subtasks)].compact.join "\n\n"
end

def format_checklist(checklist)
  checklist.map do |item|
    "* [ ] #{item}"
  end.join("\n")
end
